This is my small project for learning how to build backend for websites in python with Flask.\hypertarget{md__root_side_hoes_SomeCoolBlog_README_autotoc_md1}{}\doxysection{Features}\label{md__root_side_hoes_SomeCoolBlog_README_autotoc_md1}

\begin{DoxyItemize}
\item ability to login and create an account
\item posting and reading blog posts
\item upvoting and downvoting posts
\item comment section on posts with live feed
\end{DoxyItemize}\hypertarget{md__root_side_hoes_SomeCoolBlog_README_autotoc_md2}{}\doxysection{How to launch}\label{md__root_side_hoes_SomeCoolBlog_README_autotoc_md2}

\begin{DoxyItemize}
\item clone this git repository
\item in main directory create new python virtual environment \+: {\ttfamily python -\/m venv ./.venv}
\item activate the virtual environment\+:
\begin{DoxyItemize}
\item Linux\+: {\ttfamily sourve .venv/bin/activate}
\end{DoxyItemize}
\item install all required dependencies\+: {\ttfamily pip install -\/r requirements. txt}
\item run flask development server\+: {\ttfamily flask -\/-\/app myblog -\/-\/debug run}
\item now you can open the link that will be shown in your browser and see the website! 
\end{DoxyItemize}
